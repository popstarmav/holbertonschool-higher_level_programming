def test_add_integer():
    """
    Test the add_integer function.
    """
    # Test cases
    assert add_integer(1, 2) == 3
    assert add_integer(100, -2) == 98
    assert add_integer(2) == 100
    assert add_integer(100.3, -2) == 98

    # Test cases with invalid input
    try:
        add_integer(4, "School")
    except Exception as e:
        assert str(e) == "a must be an integer"

    try:
        add_integer(None)
    except Exception as e:
        assert str(e) == "a must be an integer"

    # Test the docstring
    docstring_lines = __import__("0-add_integer").add_integer.__doc__.strip().splitlines()
    assert len(docstring_lines) == 3

if __name__ == "__main__":
    test_add_integer()
    print("All tests passed!")
